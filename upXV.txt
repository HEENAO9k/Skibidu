-- === Blacklist ===
SaveBlacklist = function()
    if not canWriteFiles then return end
    pcall(function()
        if not isfolder(configFolder) then makefolder(configFolder) end
        local list = {}
        for word in pairs(blacklistedWords) do table.insert(list, word) end
        writefile(blacklistFile, HttpService:JSONEncode(list))
    end)
end

local function LoadBlacklist()
    if not canReadFiles then return end
    pcall(function()
        if isfile(blacklistFile) then
            local js = readfile(blacklistFile)
            local list = HttpService:JSONDecode(js)
            for _, word in ipairs(list) do blacklistedWords[word] = true end
            print("✅ โหลด Blacklist: " .. #list .. " คำ")
        end
    end)
end

local function AddToBlacklist(word)
    if not word or #word == 0 then return end
    word = word:lower()
    if blacklistedWords[word] then return end
    
    blacklistedWords[word] = true
    wrongWords[word] = true
    SaveBlacklist()
    
    Rayfield:Notify({
        Title = "🚫 Blacklist Permanent",
        Content = "คำ: " .. word:upper() .. "\nเพิ่มเข้า Blacklist ถาวร",
        Duration = 4,
    })
end

local function IsWordBlacklisted(word)
    return blacklistedWords[word:lower()] == true
end

local function GetBlacklistCount()
    local c = 0
    for _ in pairs(blacklistedWords) do c = c + 1 end
    return c
end

local function getDeltaWorkspaceFolder()
    if not deltaWorkspaceFolder or not deltaWorkspaceFolder.Parent then
        local name = "999ตัวตึง_WrongWords"
        deltaWorkspaceFolder = Workspace:FindFirstChild(name)
        if not deltaWorkspaceFolder then
            deltaWorkspaceFolder = Instance.new("Folder")
            deltaWorkspaceFolder.Name = name
            deltaWorkspaceFolder.Parent = Workspace
        end
    end
    return deltaWorkspaceFolder
end

local function SaveWrongWordToWorkspace(word)
    pcall(function()
        local folder = getDeltaWorkspaceFolder()
        if not folder:FindFirstChild(word) then
            local inst = Instance.new("StringValue")
            inst.Name = word
            inst.Value = word
            inst.Parent = folder
        end
    end)
end

local function MarkWordAsWrong(word)
    if not wrongWords[word] and word and #word > 0 then
        wrongWords[word] = true
        table.insert(wrongWordsStack, word)
        SaveWrongWordToWorkspace(word)
        if showWrongWordList and UpdateWrongWordListGui then UpdateWrongWordListGui() end
    end
end

local function LoadWrongWordsFromWorkspace()
    pcall(function()
        local folder = getDeltaWorkspaceFolder()
        wrongWords = {}
        for _, val in ipairs(folder:GetChildren()) do
            if val:IsA("StringValue") then
                wrongWords[val.Name] = true
            end
        end
        if showWrongWordList and UpdateWrongWordListGui then UpdateWrongWordListGui() end
    end)
end

local function UndoLastWrongWord()
    if #wrongWordsStack > 0 then
        local word = table.remove(wrongWordsStack)
        wrongWords[word] = nil
        pcall(function()
            local folder = getDeltaWorkspaceFolder()
            local inst = folder:FindFirstChild(word)
            if inst then inst:Destroy() end
        end)
        Rayfield:Notify({
            Title = "Undo สำเร็จ!",
            Content = "ลบการเพิ่มคำผิด: "..(word or ""),
            Duration = 2,
        })
        if showWrongWordList and UpdateWrongWordListGui then UpdateWrongWordListGui() end
    else
        Rayfield:Notify({
            Title = "Undo ไม่สำเร็จ",
            Content = "ไม่มีคำผิดใน Stack",
            Duration = 2,
        })
    end
end

local function IsWordWrong(word)
    return wrongWords[word] == true
end

local function GetWrongWordsCount()
    local c = 0
    for k,_ in pairs(wrongWords) do c = c + 1 end
    return c
end

-- === Smart Auto Chat ===
SendSmartChat = function(category)
    if not isAutoChatActive then return end
    if tick() - lastChatTime < chatCooldown then return end
    
    local messages = TauntConfig[category]
    if not messages or #messages == 0 then return end
    
    local msg = messages[math.random(1, #messages)]
    lastChatTime = tick()
    
    -- Try TextChatService (Modern)
    local success = pcall(function()
        if TextChatService.TextChannels and TextChatService.TextChannels.RBXGeneral then
            TextChatService.TextChannels.RBXGeneral:SendAsync(msg)
        end
    end)
    
    -- Fallback to Legacy
    if not success then
        pcall(function()
            local event = ReplicatedStorage:FindFirstChild("DefaultChatSystemChatEvents")
            if event then 
                local say = event:FindFirstChild("SayMessageRequest")
                if say then say:FireServer(msg, "All") end
            end
        end)
    end
end

local function RemoveESP(player)
    if not player then return end
    if PlayerBillboards[player.UserId] then
        if PlayerBillboards[player.UserId].GUI then
            PlayerBillboards[player.UserId].GUI:Destroy()
        end
        PlayerBillboards[player.UserId] = nil
    end
end

UpdatePlayerESP = function(player, statusText, statusColor, wordText)
    if not isSmartVisualsActive then return end
    if not player then return end
    
    if not PlayerBillboards[player.UserId] then
        CreateESP(player)
    end
    
    local entry = PlayerBillboards[player.UserId]
    if entry then
        if not entry.GUI.Adornee then
             if player.Character and player.Character:FindFirstChild("Head") then
                 entry.GUI.Adornee = player.Character.Head
             end
        end
        
        if statusText then 
            entry.StatusL.Text = statusText 
            if statusColor then entry.StatusL.TextColor3 = statusColor end
        end
        
        if wordText then
            entry.WordL.Text = wordText
            -- Pulse effect
            TweenService:Create(entry.WordL, TweenInfo.new(0.2, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {TextSize = 18}):Play()
            task.delay(0.2, function()
                if entry.WordL then
                     TweenService:Create(entry.WordL, TweenInfo.new(0.2), {TextSize = 16}):Play()
                end
            end)
        end
    end
end

local function ClearAllESP()
    for uid, data in pairs(PlayerBillboards) do
        if data.GUI then data.GUI:Destroy() end
    end
    PlayerBillboards = {}
    if VisualsFolder then VisualsFolder:ClearAllChildren() end
end

-- Refresh ESP loop (handles character respawn)
task.spawn(function()
    while true do
        task.wait(2)
        if isSmartVisualsActive then
            for _, player in ipairs(Players:GetPlayers()) do
                if player.Character and player.Character:FindFirstChild("Head") then
                    if not PlayerBillboards[player.UserId] then
                        CreateESP(player)
                    else
                        -- Re-attach if lost adornee
                        if not PlayerBillboards[player.UserId].GUI.Adornee then
                            PlayerBillboards[player.UserId].GUI.Adornee = player.Character.Head
                        end
                    end
                end
            end
        else
            ClearAllESP()
        end
    end
end)

local VerbFallbackList = {
    "run","walk","play","read","jump","cry","sing","draw","call","wait",
    "fail","kick","type","quit","try","test","meet","care","talk","end",
    "grow","vote","join","move","sit","send","turn","miss","stop","need"
}

local function GetTargetMinLength(prefix)
    if not isMinLengthCheckActive then return 0 end
    local n = #prefix
    if n == 2 then return 4 end
    if n == 3 then return 5 end
    return 0
end

local function IsValidWord(word)
    if not word then return false end
    if #word > maxLength then return false end
    local w = word:lower()
    if IsWordUsed(w) then return false end
    if IsWordBlacklisted(w) then return false end
    if IsWordWrong(w) then return false end
    if w == lastTypedOrSentWord then return false end
    return true
end

local function FindInSavedWords(prefix)
    local minLen = GetTargetMinLength(prefix)
    local candidates = {}
    for word, _ in pairs(VerifiedWordsDB) do
        if #word >= minLen and #word <= maxLength and word:sub(1, #prefix) == prefix then
            if IsValidWord(word) then
                table.insert(candidates, word)
                if #candidates > 30 then break end
            end
        end
    end
    if #candidates > 0 then
        -- Priority for Hard Mode in Saved Words too?
        if isHardMode then
            local hardEndings = {x=true, z=true, q=true, j=true, v=true, k=true}
            for _, w in ipairs(candidates) do
                local endChar = w:sub(-1):lower()
                if hardEndings[endChar] then
                    return w
                end
            end
        end
        return candidates[math.random(1, #candidates)]
    end
    return nil
end

local function FindSuitableVerb(prefix)
    local minLen = GetTargetMinLength(prefix)
    for _,v in ipairs(VerbFallbackList) do
        if #v >= minLen and #v <= maxLength and v:sub(1,#prefix) == prefix then
            if IsValidWord(v) then
                return v:lower()
            end
        end
    end
    return nil
end

local function FetchWordsFromAPI(prefix)
    if not prefix or #prefix == 0 then return nil end
    local minLen = GetTargetMinLength(prefix)
    
    local words = {}
    
    local url = "https://api.datamuse.com/words?sp=" .. prefix .. "*&max=20"
    local success, response = pcall(function()
        return game:HttpGet(url)
    end)
    if success then
        local data = HttpService:JSONDecode(response)
        for _, wordData in ipairs(data) do
            if wordData.word and #wordData.word >= minLen and #wordData.word <= maxLength then
                local word = wordData.word:lower()
                if not IsWordBlacklisted(word) and not IsWordWrong(word) then
                    table.insert(words, word)
                end
            end
        end
    end
    
    if #words < 5 then
        local url2 = "https://api.datamuse.com/sug?s=" .. prefix .. "&max=20"
        local success2, response2 = pcall(function() return game:HttpGet(url2) end)
        if success2 then
             local data2 = HttpService:JSONDecode(response2)
             for _, wordData in ipairs(data2) do
                if wordData.word and #wordData.word >= minLen and #wordData.word <= maxLength then
                    local word = wordData.word:lower()
                    if word:match("^[a-z]+$") and not IsWordBlacklisted(word) and not IsWordWrong(word) then
                         local exists = false
                         for _, existing in ipairs(words) do
                            if existing == word then exists = true break end
                         end
                         if not exists then table.insert(words, word) end
                    end
                end
            end
        end
    end

    if #words > 0 then
        lastAPIWords = words
        if showApiWordList and UpdateApiWordListGui then UpdateApiWordListGui(words) end
        return words
    else
        if showApiWordList and UpdateApiWordListGui then UpdateApiWordListGui({}) end
        return nil
    end
end

local function GetOpenRouterExcludedWords(prefix)
    local excluded = {}
    local seen = {}
    for word, _ in pairs(UsedWordsHistory) do
        if word:sub(1, #prefix):lower() == prefix:lower() and not seen[word] then
            table.insert(excluded, word)
            seen[word] = true
        end
    end
    for word, _ in pairs(usedAPIWords) do
        if word:sub(1, #prefix):lower() == prefix:lower() and not seen[word] then
            table.insert(excluded, word)
            seen[word] = true
        end
    end
    return excluded
end

local function GetNextOpenRouterKey()
    for _ = 1, OpenRouterKeyCount do
        openRouterKeyCursor = (openRouterKeyCursor % OpenRouterKeyCount) + 1
        local key = OpenRouterApiKeys[openRouterKeyCursor]
        if key and key:match("%S") then return key, openRouterKeyCursor end
    end
    return nil, nil
end

local function ExtractOpenRouterWordsFromParsed(parsed)
    if type(parsed) ~= "table" then return nil end
    if type(parsed.words) == "table" then return parsed.words end
    if type(parsed.data) == "table" then return parsed.data end
    if type(parsed.result) == "table" then return parsed.result end
    if #parsed > 0 then return parsed end
    return nil
end

FetchWordsFromOllama = function(prefix)
    if not prefix or #prefix == 0 then return nil end
    if not request then
        Rayfield:Notify({Title="OpenRouter Error", Content="Executor นี้ไม่รองรับ request/http_request", Duration=3})
        return nil
    end

    local minLen = GetTargetMinLength(prefix)
    local excluded = GetOpenRouterExcludedWords(prefix)
    local excludedText = (#excluded > 0) and table.concat(excluded, ", ") or "none"
    local hardText = isHardMode and "Prefer real English words ending with difficult letters: x, z, q, j, v, k. If impossible, return best valid alternatives." or "Prefer common real English words."
    local prompt = string.format([[Return ONLY valid minified JSON. Schema: {"words":["word1","word2"]}. Need exactly %d unique real English words that start with "%s", length between %d and %d, lowercase alphabetic a-z only. Do not include these already used words: %s. %s]], OpenRouterWordCount, prefix:lower(), math.max(minLen, #prefix), maxLength, excludedText, hardText)

    OpenRouterLastRawResponse = ""
    OpenRouterLastParsedContent = ""

    local words = {}
    for attempt = 1, OpenRouterKeyCount do
        local apiKey, keyIndex = GetNextOpenRouterKey()
        if not apiKey then
            Rayfield:Notify({Title="OpenRouter Error", Content="กรุณาใส่ API Key อย่างน้อย 1 ช่อง", Duration=3})
            return nil
        end

        local payload = {
            model = OpenRouterModel,
            messages = {
                {role = "system", content = "You generate word-game answers. Output JSON only. No markdown, no explanation."},
                {role = "user", content = prompt}
            },
            temperature = isHardMode and 0.35 or 0.2,
            max_tokens = 180
        }

        local success, response = pcall(function()
            return request({
                Url = OpenRouterApiUrl,
                Method = "POST",
                Headers = {
                    ["Content-Type"] = "application/json",
                    ["Authorization"] = "Bearer " .. apiKey,
                    ["HTTP-Referer"] = "https://roblox.com",
                    ["X-Title"] = "999 Word Solver"
                },
                Body = HttpService:JSONEncode(payload)
            })
        end)

        local statusCode = response and (response.StatusCode or response.Status or response.status_code)
        local body = response and (response.Body or response.body or "") or tostring(response)
        OpenRouterLastRawResponse = "KEY #" .. tostring(keyIndex) .. " | HTTP " .. tostring(statusCode) .. "\n" .. tostring(body)
        if OpenRouterDebugMode then
            print("========== OPENROUTER RAW ==========")
            print(OpenRouterLastRawResponse)
            print("========== OPENROUTER PROMPT ==========")
            print(prompt)
        end

        if success and statusCode and statusCode >= 200 and statusCode < 300 then
            local ok, data = pcall(function() return HttpService:JSONDecode(body) end)
            if ok and data and data.choices and data.choices[1] and data.choices[1].message then
                local content = data.choices[1].message.content or ""
                OpenRouterLastParsedContent = content
                content = content:gsub("```json", ""):gsub("```", ""):match("^%s*(.-)%s*$") or content
                local jsonOnly = content:match("({.*})") or content
                local okJson, parsed = pcall(function() return HttpService:JSONDecode(jsonOnly) end)
                local rawWords = okJson and ExtractOpenRouterWordsFromParsed(parsed) or nil
                if rawWords then
                    local seen = {}
                    for _, raw in ipairs(rawWords) do
                        local word = tostring(raw):lower():match("^[a-z]+$")
                        if word and word:sub(1, #prefix) == prefix:lower() and #word >= minLen and #word <= maxLength and not seen[word] and IsValidWord(word) then
                            table.insert(words, word)
                            seen[word] = true
                            if #words >= OpenRouterWordCount then break end
                        end
                    end
                    if #words > 0 then
                        lastAPIWords = words
                        if showApiWordList and UpdateApiWordListGui then UpdateApiWordListGui(words) end
                        Rayfield:Notify({Title="OpenRouter AI", Content="ใช้ Key #"..tostring(keyIndex).." ได้ "..#words.." คำ", Duration=2})
                        return words
                    end
                elseif OpenRouterDebugMode then
                    warn("OpenRouter JSON parse failed. AI content: " .. tostring(content))
                end
            elseif ok and data and data.error then
                warn("OpenRouter API error: " .. tostring(data.error.message or data.error))
            end
        else
            warn("OpenRouter key #" .. tostring(keyIndex) .. " failed: " .. tostring(statusCode) .. " " .. tostring(body))
        end
    end

    if showApiWordList and UpdateApiWordListGui then UpdateApiWordListGui({}) end
    return nil
end

local function GetUnusedAPIWord()
    if not lastAPIWords or #lastAPIWords == 0 then return nil end
    
    -- HARD MODE LOGIC (Local Datamuse)
    if isHardMode then
        local hardEndings = {x=true, z=true, q=true, j=true, v=true, k=true}
        -- First pass: Try to find a hard ending word
        for _, word in ipairs(lastAPIWords) do
            if not usedAPIWords[word] and IsValidWord(word) then
                local lastChar = word:sub(-1):lower()
                if hardEndings[lastChar] then
                    usedAPIWords[word] = true
                    Rayfield:Notify({Title="HARD MODE 🔥", Content="เจอคำยาก: "..word, Duration=1})
                    return word
                end
            end
        end
        -- If none found, fall through to normal logic
    end

    if apiLeastPickMode then
        local shortest = nil
        local shortestLen = 9999
        local minLenCfg = tonumber(apiLeastPickMinLen) or 0
        
        for _, word in ipairs(lastAPIWords) do
            if not usedAPIWords[word] and IsValidWord(word) then
                if #word >= minLenCfg then
                    if #word < shortestLen then
                        shortest = word
                        shortestLen = #word
                    end
                end
            end
        end
        
        if shortest then
            usedAPIWords[shortest] = true
            return shortest
        end
        return nil
    else
        for _, word in ipairs(lastAPIWords) do
            if not usedAPIWords[word] and IsValidWord(word) then
                usedAPIWords[word] = true
                return word
            end
        end
        return nil
    end
end

local function findWordFromDatabase(prefix)
    local minLen = GetTargetMinLength(prefix)
    local candidates = {}
    for _, word in ipairs(WordDatabase) do
        if #word >= minLen and #word <= maxLength and word:sub(1, #prefix) == prefix then
            if IsValidWord(word) then
                table.insert(candidates, word)
                if #candidates >= 50 then break end
            end
        end
    end
    
    -- Hard Mode Filter for Database
    if isHardMode and #candidates > 0 then
         local hardEndings = {x=true, z=true, q=true, j=true, v=true, k=true}
         local hardCandidates = {}
         for _, w in ipairs(candidates) do
             local endChar = w:sub(-1):lower()
             if hardEndings[endChar] then
                 table.insert(hardCandidates, w)
             end
         end
         if #hardCandidates > 0 then
             return hardCandidates[math.random(1, #hardCandidates)]
         end
    end

    return #candidates > 0 and candidates[math.random(1, #candidates)] or nil
end

local function TryGetWordFromSource(sourceName, prefix)
    if sourceName == "None" then return nil end
    
    if sourceName == "Saved Words (Persistent)" then
        return FindInSavedWords(prefix), "saved"
    elseif sourceName == "API (Datamuse)" then
        local unused = GetUnusedAPIWord()
        if unused then return unused, "api" else return nil end
    elseif sourceName == "OpenRouter AI" then
        local ollamaWords = FetchWordsFromOllama(prefix)
        if ollamaWords and #ollamaWords > 0 then
            -- OpenRouter JSON response is validated and sorted by model preference
            local pick = ollamaWords[1] -- Pick the first one as it's sorted by priority
            usedAPIWords[pick] = true
            return pick, "openrouter_ai"
        end
        return nil
    elseif sourceName == "Database (Built-in)" then
        local w = findWordFromDatabase(prefix)
        if w then return w, "database" end
    elseif sourceName == "Custom DB (Amnexe)" then
        local w = findWordFromCustomDatabase(prefix)
        if w then return w, "custom_db" end
    end
    return nil
end

findWord = function(prefix)
    local minLen = GetTargetMinLength(prefix)
    local resultWord = nil
    local resultSource = nil
    
    if useSavedWordsFirst then
        local saved = FindInSavedWords(prefix)
        if saved then
            if showDecisionLog then UpdateDecisionLogGui(prefix, minLen, saved.." (Saved Force)") end
            return saved, "saved"
        end
    end

    for i = 1, 3 do
        local source = SourcePriorities[i]
        if source and source ~= "None" then
            local word, src = TryGetWordFromSource(source, prefix)
            if word then
                resultWord = word
                resultSource = src
                break
            end
        end
    end
    
    if not resultWord then
        local verb = FindSuitableVerb(prefix)
        if verb then
            resultWord = verb
            resultSource = "verb_fallback"
        end
    end
    
    if showDecisionLog then UpdateDecisionLogGui(prefix, minLen, resultWord or "None") end
    return resultWord, resultSource
end

extractWordFromGame = function()
    local inGame = LocalPlayer.PlayerGui:FindFirstChild("InGame")
    if not inGame then return nil end
    local frame = inGame:FindFirstChild("Frame")
    if not frame then return nil end
    local currentWordFolder = frame:FindFirstChild("CurrentWord")
    if not currentWordFolder then return nil end

    local letters = {}
    for _, child in pairs(currentWordFolder:GetChildren()) do
        local num = tonumber(child.Name)
        if num then
            local letter = child:FindFirstChild("Letter")
            if letter and letter:IsA("TextLabel") then
                letters[num] = letter.Text
            end
        end
    end

    local sortedLetters = {}
    for i = 1, 20 do
        if letters[i] then table.insert(sortedLetters, letters[i]) end
    end
    if #sortedLetters == 0 then return nil end
    local word = table.concat(sortedLetters, ""):lower()
    return word
end

checkIfMyTurn = function()
    local success, result = pcall(function()
        local inGame = LocalPlayer.PlayerGui:FindFirstChild("InGame")
        if inGame then
            local frame = inGame:FindFirstChild("Frame")
            if frame then
                local typeObj = frame:FindFirstChild("Type")
                if typeObj and (typeObj:IsA("TextLabel") or typeObj:IsA("TextBox") or typeObj:IsA("TextButton")) then
                    local text = typeObj.Text
                    if text:find(LocalPlayer.Name) then
                        return true, text
                    end
                end
            end
        end
        return false, ""
    end)
    if success then return result end
    return false, ""
end

    local inGame = LocalPlayer.PlayerGui:FindFirstChild("InGame")
    if inGame then
        local frame = inGame:FindFirstChild("Frame")

                end

        end
    end
    return nil
end

local function getRandomDelay(baseSpeed)
    if isHumanLikeTyping then
        local variance = math.random() * 0.05
        local delay = baseSpeed + variance
        if math.random() < 0.05 then
            delay = delay + math.random(0.1, 0.3)
        end
        return math.max(0.015, delay)
    else
        return math.max(0.008, baseSpeed * (0.5 + math.random() * 1))
    end
end

deleteAllText = function()
    local deleteCount = math.max(lastTypedLength, 5)
    for i = 1, deleteCount do
        pcall(function()
            VirtualInputManager:SendKeyEvent(true, Enum.KeyCode.Backspace, false, game)
            task.wait(0.01)
            VirtualInputManager:SendKeyEvent(false, Enum.KeyCode.Backspace, false, game)
        end)
        task.wait(getRandomDelay(deleteSpeed))
    end
    lastTypedLength = 0
end

local function CheckHighlightColor()
    local Character = LocalPlayer.Character
    if not Character then return nil end

    for _, obj in pairs(Character:GetDescendants()) do
        if obj:IsA("Highlight") and obj.Enabled then
            local isInvisible = (obj.FillTransparency >= 0.95) and (obj.OutlineTransparency >= 0.95)
            
            if not isInvisible then
                local hColor = obj.FillColor
                local r, g, b = hColor.R, hColor.G, hColor.B
                
                if r > UI_CONFIG.RedThreshold and g < 0.4 then
                    return "red"
                elseif g > UI_CONFIG.GreenThreshold and r < 0.4 then
                    return "green"
                else
                    return "other"
                end
            end
        end
    end
    
    return nil
end

local function CheckOtherPlayerHighlight(playerName)
    local player = Players:FindFirstChild(playerName)
    if not player or not player.Character then return nil end
    
    for _, obj in pairs(player.Character:GetDescendants()) do
        if obj:IsA("Highlight") and obj.Enabled then

            if not isInvisible then
                local hColor = obj.FillColor
                local r, g, b = hColor.R, hColor.G, hColor.B
                if g > UI_CONFIG.GreenThreshold and r < 0.4 then
                    return "green"
                end
            end
        end
    end
    return nil
end

StartHighlightDetection = function(wordToCheck)
    StopHighlightDetection() 
    
    highlightDetectionActive = true
    detectionStartTime = tick()
    
    highlightConnection = RunService.RenderStepped:Connect(function()
        if not highlightDetectionActive then return end
        
        -- Timeout logic
        if tick() - detectionStartTime > DETECTION_TIMEOUT then
            StopHighlightDetection()
            isProcessing = false -- Release lock on timeout
            isWaitingForOtherPlayer = true
            return
        end
        
        -- NEW FEATURE: Check for "Already Used" (No Highlight but Stuck)
        if tick() - detectionStartTime > DETECTION_STUCK_TIMEOUT then
             local isMyTurn, _ = checkIfMyTurn()
             if isMyTurn then
                 -- Stuck! Word was likely rejected without visual feedback (Already Used)
                 StopHighlightDetection()
                 MarkWordAsWrong(wordToCheck) -- Temporarily mark as wrong so we don't pick it again immediately
                 
                 Rayfield:Notify({
                     Title = "⚠️ ตรวจพบคำซ้ำ (Already Used)",
                     Content = "ระบบจะลบและหาคำใหม่...",
                     Duration = 2,
                     Image = 4483362458
                 })
                 
                 -- Smart Retry Logic
                 task.spawn(function()
                     deleteAllText()
                     task.wait(0.1)
                     local extracted = extractWordFromGame()
                     if extracted then
                         local newWord, newSource = findWord(extracted)
                         if newWord and newWord ~= wordToCheck then
                             -- Retry with new word
                             autoType(extracted, newWord, true, newSource)
                         else
                             isProcessing = false
                             isWaitingForOtherPlayer = true
                         end
                     else
                         isProcessing = false
                         isWaitingForOtherPlayer = true
                     end
                 end)
                 return
             end
        end

        local color = CheckHighlightColor()
        
        if color == "red" then
            StopHighlightDetection()
            AddToBlacklist(wordToCheck)
            MarkWordAsWrong(wordToCheck)
            
            -- Evolution: Auto Chat
            SendSmartChat("Fail")
            -- Evolution: Visuals
            UpdatePlayerESP(LocalPlayer, "WRONG!", Color3.fromRGB(255, 50, 50), wordToCheck:upper())
            
            Rayfield:Notify({
                Title = "🔴 ตรวจพบคำผิด (Highlight)!",
                Content = "คำ: " .. wordToCheck:upper() .. "\n🚫 เพิ่มเข้า Blacklist ถาวร",
                Duration = 4,
            })
            
            -- Panic Fix: Handle re-typing safely (isProcessing stays true)
            task.spawn(function()
                task.wait(0.5)
                deleteAllText()
                task.wait(0.2)
                local extracted = extractWordFromGame()
                if extracted and #extracted > 0 then
                    local newWord, newSource = findWord(extracted)
                    if newWord and newWord ~= wordToCheck then
                        autoType(extracted, newWord, true, newSource)
                    else
                        isProcessing = false
                        isWaitingForOtherPlayer = true
                    end
                else
                    isProcessing = false
                    isWaitingForOtherPlayer = true
                end
            end)
            
        elseif color == "green" then
            StopHighlightDetection()
            MarkWordAsUsed(wordToCheck)
            AddToVerifiedWords(wordToCheck)
            
            -- Evolution: Auto Chat
            SendSmartChat("Success")
            -- Evolution: Visuals
            UpdatePlayerESP(LocalPlayer, "CORRECT!", Color3.fromRGB(50, 255, 50), wordToCheck:upper())
            
            Rayfield:Notify({
                Title = "✅ คำถูกต้อง!",
                Content = "คำ: " .. wordToCheck:upper() .. "\n💾 บันทึกแล้ว",
                Duration = 2,
            })

            -- Strict delay logic: Wait 1.5s before releasing lock
            task.wait(1.5)
            isProcessing = false
            isWaitingForOtherPlayer = true
        end
    end)
end

autoType = function(prefix, fullWord, isAutoMode, source)
    StopHighlightDetection()
    
    local remaining = fullWord:sub(#prefix + 1)
    lastTypedLength = #fullWord
    lastTypedOrSentWord = fullWord
    lastTypedOrSentWordTime = tick()
    
    local thisRunId = HttpService:GenerateGUID(false)
    activeRunId = thisRunId
    
    task.spawn(function()
        -- NEW: START DELAY LOGIC
        if startDelay > 0 then
            UpdatePlayerESP(LocalPlayer, "WAITING...", Color3.fromRGB(255, 200, 0), "Delay: " .. startDelay .. "s")
            task.wait(startDelay)
        end
        
        -- Check if interrupted during delay
        if activeRunId ~= thisRunId then return end
        
        -- Evolution: Update Local Visuals
        UpdatePlayerESP(LocalPlayer, "TYPING...", Color3.fromRGB(50, 150, 255), fullWord:upper())
        
        for i = 1, #remaining do
            if activeRunId ~= thisRunId then return end
            local char = remaining:sub(i,i):upper()
            
            -- FIX FOR QWERTZ LAYOUT (Y <-> Z SWAP)
            local codeToSend = Enum.KeyCode[char]
            if char == "Y" then
                codeToSend = Enum.KeyCode.Z
            elseif char == "Z" then
                codeToSend = Enum.KeyCode.Y
            end
            
            pcall(function()
                VirtualInputManager:SendKeyEvent(true, codeToSend, false, game)
                if not isSuperFastMode then -- SUPER FAST MODE CHECK
                    task.wait(0.01)
                end
                VirtualInputManager:SendKeyEvent(false, codeToSend, false, game)
            end)
            
            if not isSuperFastMode then -- SUPER FAST MODE CHECK
                task.wait(getRandomDelay(baseTypeSpeed))
            end
        end
        
        if activeRunId ~= thisRunId then return end
        
        if ErrorDetectionMethod == "highlight" then
            StartHighlightDetection(fullWord)
        end
        
        if not isSuperFastMode then -- SUPER FAST MODE CHECK
            task.wait(getRandomDelay(0.1))
        end
        if activeRunId ~= thisRunId then return end
        
        VirtualInputManager:SendKeyEvent(true, Enum.KeyCode.Return, false, game)
        if not isSuperFastMode then -- SUPER FAST MODE CHECK
            task.wait(0.03)
        end
        VirtualInputManager:SendKeyEvent(false, Enum.KeyCode.Return, false, game)
        
        lastTypedOrSentWord = fullWord
        lastTypedOrSentWordTime = tick()

        local badge = ({api="🌐", database="💾", verb_fallback="⚡", custom_db="📂", saved="⭐", openrouter_ai="🤖"})[source or ""] or ""
        local srcName = ({
            api="API", 
            database="Database", 
            verb_fallback="VERB Fallback", 
            saved="Saved Words",
            custom_db="Custom DB",
            openrouter_ai="OpenRouter AI"
        })[source or ""] or "?"
        
        Rayfield:Notify({
            Title = "พิมพ์สำเร็จ! " .. badge,
            Content = string.format("คำ: %s\nที่มา: %s\nความยาว: %d\n✨ รอผล Highlight...",
                fullWord:upper(), srcName, #fullWord),
            Duration = 3,
            Image = 4483362458,
        })
        
        -- Evolution: Visual Update after typing
        UpdatePlayerESP(LocalPlayer, "SENT", Color3.fromRGB(200, 200, 255), fullWord:upper())

        -- If highlight detection is on, we surrender control to StartHighlightDetection
        if ErrorDetectionMethod == "highlight" then
            return 
        end

        -- Old Logic Fallback (If detection method is NOT highlight)
        if isAutoMode then
            task.wait(1)
            if activeRunId ~= thisRunId then return end
            
            local stillMyTurn, _ = checkIfMyTurn()
            if stillMyTurn then
                apiAttemptCount = apiAttemptCount + 1
                deleteAllText()
                task.wait(0.2)
                local extracted = extractWordFromGame()
                if extracted and #extracted > 0 then
                    local newWord, newSource = findWord(extracted)
                    if newWord and newWord ~= fullWord then
                        autoType(extracted, newWord, true, newSource)
                    else
                        isProcessing = false
                        isWaitingForOtherPlayer = true
                        apiAttemptCount = 0
                        usedAPIWords = {}
                    end
                else
                    isProcessing = false
                    isWaitingForOtherPlayer = true
                    apiAttemptCount = 0

local function StartAutoPlay()
    if autoPlayConnection then return end
    Rayfield:Notify({
        Title = "Auto Play เริ่มทำงาน! 🤖",
        Content = "กำลังตรวจจับรอบของคุณ...",
        Duration = 4,
        Image = 4483362458,
    })
    
    -- Also start other player detection if configured
    if isOtherPlayerDetectionActive then
        StartOtherPlayerDetection()
    end
    
    -- Evolution: Greeting
    SendSmartChat("Greeting")
    
    autoPlayConnection = RunService.Heartbeat:Connect(function()
        if not isAutoPlayActive or isProcessing then return end
        
        for i=1,3 do
            local s = SourcePriorities[i]
            if s == "Database (Built-in)" and not isDBLoaded then return end
        end

        local isMyTurn, _ = checkIfMyTurn()
        if isMyTurn then
            if isWaitingForOtherPlayer and lastDetectedPlayer == LocalPlayer.Name then
                return
            elseif isWaitingForOtherPlayer then
                isWaitingForOtherPlayer = false
            end
            lastDetectedPlayer = LocalPlayer.Name
            isProcessing = true
            apiAttemptCount = 0
            usedAPIWords = {}
            
            -- Evolution: Visuals
            UpdatePlayerESP(LocalPlayer, "AUTO PLAY", Color3.fromRGB(0, 255, 255), "...")
            
            -- IMPROVED: Faster Real-time polling
            local extracted = nil
            for _ = 1, 10 do -- Poll for up to 0.5s but break early
                extracted = extractWordFromGame()
                if extracted and #extracted > 0 then break end
                task.wait(0.05)
            end

            if extracted and #extracted > 0 then
                local needApi = false
                for i=1,3 do if SourcePriorities[i] == "API (Datamuse)" then needApi = true break end end
                if needApi then lastAPIWords = FetchWordsFromAPI(extracted) end

                local foundWord, source = findWord(extracted)
                if foundWord then
                    autoType(extracted, foundWord, true, source)
                else
                    Rayfield:Notify({
                        Title = "ไม่พบคำ ❌",
                        Content = "บันทึกคำผิดแบบอัตโนมัติ หากพบ Highlight",
                        Duration = 2,
                    })
                    -- Even if no word found, wait for other player or manual action to clear flag
                    isProcessing = false
                    isWaitingForOtherPlayer = true
                end
            else
                isProcessing = false
                isWaitingForOtherPlayer = true
            end
        else
            if isWaitingForOtherPlayer then isWaitingForOtherPlayer = false end
        end
    end)
end

    isWaitingForOtherPlayer = false
    lastDetectedPlayer = ""
    Rayfield:Notify({

        Content = "กลับสู่โหมดปกติ",

local function getFullPath(obj)
    local path = obj.Name
    local parent = obj.Parent
    while parent and parent ~= game do
        path = parent.Name .. "/" .. path
        parent = parent.Parent
    end
    return path
end

local interestClasses = {
    "RemoteEvent", "RemoteFunction", "ModuleScript",
    "Tool", "Script", "LocalScript", "Model", "Folder"
}

local function isInteresting(obj)
    for _, class in ipairs(interestClasses) do
        if obj:IsA(class) then
            return true
        end
    end
    return false
end

local function scanAndLogContainer(container, containerName)
    local results = {}
    for _, obj in ipairs(container:GetDescendants()) do
        if isInteresting(obj) then
            local line = string.format("[%s] %s | Class: %s | Path: %s",
                containerName, obj.Name, obj.ClassName, getFullPath(obj))
            table.insert(results, line)
        end
    end
    return results
end

local function ScanWorkspaceAndRS()
    if not canWriteFiles then
        Rayfield:Notify({
            Title = "ผิดพลาด ⚠️",
            Content = "Executor ไม่รองรับ writefile",
            Duration = 4,
        })
        return
    end
    
    Rayfield:Notify({
        Title = "กำลังสแกน... 🔍",
        Content = "กำลังวิเคราะห์ Workspace + ReplicatedStorage",
        Duration = 3,
    })
    
    local containers = {
        {name = "Workspace", object = workspace},
        {name = "ReplicatedStorage", object = ReplicatedStorage}
    }
    
    local logData = {}
    
    for _, item in ipairs(containers) do
        local containerName = item.name
        local container = item.object
        if container then
            local result = scanAndLogContainer(container, containerName)
            for _, line in ipairs(result) do
                table.insert(logData, line)
            end
        end
    end
    
    local filename = "workspace_and_rs_log.txt"
    local success = pcall(function()
        writefile(filename, table.concat(logData, "\n"))
    end)
    
    if success then
        Rayfield:Notify({
            Title = "สแกนสำเร็จ! ✅",
            Content = string.format("บันทึก %d รายการที่: %s", #logData, filename),
            Duration = 5,
            Image = 4483362458,
        })
    else
        Rayfield:Notify({
            Title = "ผิดพลาด ❌",
            Content = "ไม่สามารถบันทึกไฟล์ได้",
            Duration = 3,
        })
    end
end
